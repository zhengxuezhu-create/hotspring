plot_gam <- function(data, xvar, xlabel, lty = "solid") {

      method = "gam",

      color = "#E69F00",
      linewidth = 1.2,
      linetype = lty # Elegant way to handle line types
    ) +
    labs(x = xlabel, y = "Species Richness") +

extract_gam_stats <- function(gam_model) {

  data.frame(
    edf      = s$s.table[1, "edf"],
    chi_sq   = s$s.table[1, "Chi.sq"],
    p_value  = s$s.table[1, "p-value"],
    dev_expl = s$dev.expl
  )
}

get_stars <- function(p) {
  case_when(
    p < 0.001 ~ "***",
    p < 0.01  ~ "**",
    p < 0.05  ~ "*",
    TRUE      ~ ""
  )
}

edges <- coef_table %>%
  transmute(
    from     = Predictor,
    to       = Response,
    est      = as.numeric(Std.Estimate),
    p        = as.numeric(P.Value),
    label    = sprintf("%.2f%s", est, get_stars(p)),
    sig      = p < 0.05,
    color    = ifelse(sig, "#E69F00", "gray80"),
    linetype = ifelse(sig, "solid", "dashed"),
    size     = ifelse(sig, ifelse(p < 0.001, 1.2, 0.9), 0.5)
  )

# ---------- Node layout ----------
rank_blocks <- "
  { rank = same; pH_soil; Soil_m; Temp_soil }
  { rank = same; pH_cv }
  { rank = same; LITTER }
  { rank = same; CRUST }
  { rank = same; richnesstotal }
"

levels_raw <- str_extract_all(rank_blocks, "(?<=rank = same; ).*?(?=\\s*\\})")[[1]]
levels <- str_split(levels_raw, ";\\s*") %>% lapply(str_trim)

all_nodes <- unique(c(edges$from, edges$to))
missing_nodes <- setdiff(all_nodes, unlist(levels))
if (length(missing_nodes) > 0) levels <- c(levels, list(missing_nodes))

n_levels <- length(levels)
node_pos <- map_dfr(seq_along(levels), function(i) {
  nodes <- levels[[i]]
  y <- n_levels - i + 1
  n <- length(nodes)
  x <- seq(-(n-1)/2, (n-1)/2, length.out = n) * 2.5
  tibble(name = nodes, x = x, y = y)
})

# ---------- Edge coordinates (fixed grouping) ----------
edges_pos <- edges %>%
  left_join(node_pos, by = c("from" = "name")) %>%
  rename(from_x = x, from_y = y) %>%
  left_join(node_pos, by = c("to" = "name")) %>%
  rename(to_x = x, to_y = y) %>%
  mutate(pair = paste(pmin(from, to), pmax(from, to), sep = "|")) %>%  # build column first
  group_by(pair) %>%
  mutate(
    curve_base = ifelse(from < to, 0.15, -0.15),
    curvature  = curve_base + (row_number() - 1) * 0.12 * sign(curve_base)
  ) %>%
  ungroup() %>%
  mutate(
    label_x = (from_x + to_x) / 2,
    label_y = (from_y + to_y) / 2,
    nudge_y = 0.15 * ifelse(from_y > to_y, 1, -1)
  )

# Extract standardized coefficients
std <- coefs(sem_model_updated, standardize = "scale")
path_direct <- std %>%
  dplyr::select(Response, Predictor, Std.Estimate)

# Helper to extract a direct path coefficient (returns 0 if path not present)
get_std <- function(pred, resp, data = path_direct) {
  val <- data %>%
    filter(Predictor == pred, Response == resp) %>%
    pull(Std.Estimate)
  if (length(val) == 0) return(0)
  return(val)
}

# Helper for indirect effect through a mediator
get_indirect <- function(pred, mediator, outcome, data = path_direct) {
  get_std(pred, mediator, data) * get_std(mediator, outcome, data)
}
